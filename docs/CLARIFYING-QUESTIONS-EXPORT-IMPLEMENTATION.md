# Clarifying Questions Export — Implementation Document

> Export AI-generated clarifying questions as branded, formatted documents with full template support.

---

## 📋 Overview

This feature allows users to export clarifying questions generated by the Q&A Engagement Tools as professional documents (PDF, DOCX, etc.) with organization branding and template support.

### Key Capabilities
- Export button in Q&A Engagement Tools tab
- Filter questions by status (exclude dismissed, submitted-only, etc.)
- Full template support for branding (logos, fonts, colors, headers/footers)
- CSS class-based styling — templates override default styles
- Creates an RFP Document that appears in the Documents hub
- Editable in TipTap editor before final export
- Regenerate document when questions change

### Affected Packages
- `packages/core` — Schema updates
- `apps/functions` — New helper + conditional in existing worker
- `apps/web` — Export button in Q&A Engagement Panel

### Architecture Decision: No Separate Worker

We reuse the existing `generate-document-worker.ts` with a conditional branch for `CLARIFYING_QUESTIONS`. This avoids:
- Additional SQS queue
- Additional Lambda + CloudWatch Log Group
- CDK stack complexity

The worker detects `documentType === 'CLARIFYING_QUESTIONS'` and calls the helper directly (no AI generation).

```typescript
// ─── Filter & Sort ───

export const filterQuestions = (
  questions: ClarifyingQuestionItem[],
  options: { excludeStatuses?: ClarifyingQuestionStatus[]; includeStatuses?: ClarifyingQuestionStatus[] },
): ClarifyingQuestionItem[] => {
  if (options.includeStatuses?.length) {
    return questions.filter(q => options.includeStatuses!.includes(q.status));
  }
  const excludeSet = new Set(options.excludeStatuses ?? ['DISMISSED']);
  return questions.filter(q => !excludeSet.has(q.status));
};

export const sortQuestions = (
  questions: ClarifyingQuestionItem[],
  sortBy: string,
  sortOrder: string,
): ClarifyingQuestionItem[] => {
  const sorted = [...questions].sort((a, b) => {
    switch (sortBy) {
      case 'priority':
        return PRIORITY_ORDER[a.priority] - PRIORITY_ORDER[b.priority];
      case 'category':
        return CATEGORY_ORDER[a.category] - CATEGORY_ORDER[b.category];
      case 'status':
        return a.status.localeCompare(b.status);
      case 'createdAt':
      default:
        return new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime();
    }
  });
  return sortOrder === 'desc' ? sorted.reverse() : sorted;
};

// ─── HTML Generation (CSS Classes) ───

const groupByCategory = (
  questions: ClarifyingQuestionItem[],
): Map<string, ClarifyingQuestionItem[]> => {
  const groups = new Map<string, ClarifyingQuestionItem[]>();
  for (const q of questions) {
    const list = groups.get(q.category) ?? [];
    list.push(q);
    groups.set(q.category, list);
  }
  return groups;
};

const buildQuestionHtml = (
  q: ClarifyingQuestionItem,
  index: number,
  options: { includeRationale: boolean; includeReferences: boolean },
): string => {
  const priorityClass = `cq-priority-${q.priority.toLowerCase()}`;
  
  let html = `
    <div class="cq-question">
      <div class="cq-question-header">
        <span class="cq-question-number">${index}.</span>
        <span class="cq-question-text">${q.question}</span>
        <span class="cq-priority-badge ${priorityClass}">${q.priority}</span>
      </div>
  `;
  
  if (options.includeRationale && q.rationale) {
    html += `<p class="cq-rationale"><strong>Why ask:</strong> ${q.rationale}</p>`;
  }
  
  if (options.includeReferences && q.ambiguitySource?.snippet) {
    html += `
      <div class="cq-reference">
        <strong>RFP Reference:</strong> "${q.ambiguitySource.snippet}"
        ${q.ambiguitySource.sectionRef ? `<p class="cq-reference-section">Section: ${q.ambiguitySource.sectionRef}</p>` : ''}
      </div>
    `;
  }
  
  html += '</div>';
  return html;
};

export const buildQuestionsHtml = (
  questions: ClarifyingQuestionItem[],
  options: {
    groupByCategory: boolean;
    includeRationale: boolean;
    includeReferences: boolean;
  },
): string => {
  if (options.groupByCategory) {
    const groups = groupByCategory(questions);
    const sections: string[] = [];
    let globalIndex = 1;
    
    for (const [category, categoryQuestions] of groups) {
      const categoryLabel = category.charAt(0) + category.slice(1).toLowerCase();
      const categoryHtml = `
        <div class="cq-category-section">
          <h2 class="cq-category-title">${categoryLabel} Questions</h2>
          ${categoryQuestions.map(q => buildQuestionHtml(q, globalIndex++, options)).join('\n')}
        </div>
      `;
      sections.push(categoryHtml);
    }
    return sections.join('\n');
  }
  
  return questions.map((q, i) => buildQuestionHtml(q, i + 1, options)).join('\n');
};

 * Used when no template is available.

// ─── Main Generation Function ───

export interface GenerateClarifyingQuestionsDocumentParams {
  orgId: string;
  projectId: string;
  opportunityId: string;
  documentId: string;
  templateId?: string;
  options?: ClarifyingQuestionsExportOptions;
}

export const generateClarifyingQuestionsDocument = async (
  params: GenerateClarifyingQuestionsDocumentParams,
): Promise<void> => {
  const { orgId, projectId, opportunityId, documentId, templateId, options } = params;
  const effectiveOptions = options ?? {};

  console.log(`Processing clarifying questions document for documentId=${documentId}`);

  // 1. Load clarifying questions
  const allQuestions = await getClarifyingQuestions(orgId, projectId, opportunityId);
  
  if (!allQuestions.length) {
    await updateDocumentStatus(
      projectId, opportunityId, documentId, 'FAILED',
      undefined, 'No clarifying questions found for this opportunity',
    );
    return;
  }

  // 2. Filter and sort questions
  const filtered = filterQuestions(allQuestions, effectiveOptions);
  
  if (!filtered.length) {
    await updateDocumentStatus(
      projectId, opportunityId, documentId, 'FAILED',
      undefined, 'No questions match the selected filter criteria',
    );
    return;
  }

  const sorted = sortQuestions(
    filtered,
    effectiveOptions.sortBy ?? 'priority',
    effectiveOptions.sortOrder ?? 'desc',
  );

  console.log(`Filtered ${sorted.length} questions from ${allQuestions.length} total`);

  // 3. Build macro values and resolve template
  const macroValues = await buildMacroValues({ orgId, projectId, opportunityId });
  const templateHtml = await resolveTemplateHtml(orgId, 'CLARIFYING_QUESTIONS', templateId, macroValues);

  // 4. Generate document HTML
  let finalHtml: string;

  if (templateHtml) {
    // Template provides structure + its own <style> block to override defaults
    // Inject questions HTML into {{CONTENT}} placeholder
    const questionsHtml = buildQuestionsHtml(sorted, {
      groupByCategory: effectiveOptions.groupByCategory ?? true,
      includeRationale: effectiveOptions.includeRationale ?? true,
      includeReferences: effectiveOptions.includeReferences ?? true,
    });
    // Template should include its own <style> block to style .cq-* classes
    finalHtml = replaceMacros(templateHtml, { ...macroValues, CONTENT: questionsHtml });
  } else {
    // No template — use default formatting with default styles
    finalHtml = buildDefaultDocumentHtml(sorted, effectiveOptions, macroValues);
  }

  // 5. Save the document
  await updateDocumentStatus(
    projectId, opportunityId, documentId, 'COMPLETE',
    { title: 'Clarifying Questions', content: finalHtml },
    undefined, orgId,
  );

  console.log(`Clarifying questions document complete for documentId=${documentId}: ${sorted.length} questions`);
};
```

### Default Behavior (No Template)

When no template exists for `CLARIFYING_QUESTIONS`, the document uses `DEFAULT_CLARIFYING_QUESTIONS_STYLES`:


### With Template (Branding Override)

Organization creates a `CLARIFYING_QUESTIONS` template with their own `<style>` block:

```html

  .cq-priority-high { background: #cc0000; /* brand red */ }
  .cq-category-title { color: #0066cc; border-color: #0066cc; }
</style>

<header style="text-align: center; margin-bottom: 2em;">
  <img src="{{LOGO_URL}}" style="max-width: 200px;" />
  <p>{{COMPANY_NAME}}</p>
</header>

<h1 class="cq-title">Clarifying Questions</h1>
<p>For: {{OPPORTUNITY_TITLE}}</p>
<p>Solicitation: {{SOLICITATION_NUMBER}}</p>

{{CONTENT}}

<footer style="margin-top: 3em; text-align: center; font-size: 0.8em;">
  <p>{{COMPANY_NAME}} | Confidential | {{TODAY}}</p>
</footer>
```

The template's `<style>` block overrides the defaults because:
1. CSS cascade — later styles win
2. Both use the same `.cq-*` class names
3. Template can add new styles for elements outside `{{CONTENT}}`

### 1. Hook: `apps/web/components/qa-engagement/hooks/useExportClarifyingQuestions.ts`

```typescript
'use client';

import { useState } from 'react';
import { apiMutate } from '@/lib/api';
import type { ClarifyingQuestionsExportOptions } from '@auto-rfp/core';

interface ExportResult {
  documentId: string;
  status: string;
}

export const useExportClarifyingQuestions = () => {
  const [isExporting, setIsExporting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const exportQuestions = async (
    orgId: string,
    projectId: string,
    opportunityId: string,
    options?: ClarifyingQuestionsExportOptions,
    templateId?: string,
  ): Promise<ExportResult | null> => {
    setIsExporting(true);
    setError(null);

    try {
      const result = await apiMutate<ExportResult>('/rfp-document/generate', {
        method: 'POST',
        body: {
          projectId,
          opportunityId,
          documentType: 'CLARIFYING_QUESTIONS',
          templateId,
          options,
        },
      });
      return result;
    } catch (err) {
      const message = err instanceof Error ? err.message : 'Failed to export questions';
      setError(message);
      return null;
    } finally {
      setIsExporting(false);
    }
  };

  return { exportQuestions, isExporting, error };
};
```

```tsx
          <div className="space-y-2">
            <Label className="text-sm font-medium">Sort By</Label>
            <Select value={sortBy} onValueChange={(v) => setSortBy(v as typeof sortBy)}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="priority">Priority (High → Low)</SelectItem>
                <SelectItem value="category">Category</SelectItem>
                <SelectItem value="createdAt">Date Created</SelectItem>
              </SelectContent>
            </Select>
          </div>

          {/* Content Options */}
          <div className="space-y-2">
            <Label className="text-sm font-medium">Content Options</Label>
            <div className="space-y-2">
              <div className="flex items-center space-x-2">
                <Checkbox
                  id="group-category"
                  checked={groupByCategory}
                  onCheckedChange={(checked) => setGroupByCategory(checked as boolean)}
                />
                <label htmlFor="group-category" className="text-sm">
                  Group by category
                </label>
              </div>
              <div className="flex items-center space-x-2">
                <Checkbox
                  id="include-rationale"
                  checked={includeRationale}
                  onCheckedChange={(checked) => setIncludeRationale(checked as boolean)}
                />
                <label htmlFor="include-rationale" className="text-sm">
                  Include rationale (why ask this question)
                </label>
              </div>
              <div className="flex items-center space-x-2">
                <Checkbox
                  id="include-references"
                  checked={includeReferences}
                  onCheckedChange={(checked) => setIncludeReferences(checked as boolean)}
                />
                <label htmlFor="include-references" className="text-sm">
                  Include RFP references
                </label>
              </div>
            </div>
          </div>
        </div>

        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>
            Cancel
          </Button>
          <Button onClick={handleExport} disabled={isExporting}>
            {isExporting ? (
              <>
                <Loader2 className="h-4 w-4 mr-2 animate-spin" />
                Creating...
              </>
            ) : (
              <>
                <FileDown className="h-4 w-4 mr-2" />
                Create Document
              </>
            )}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
};
```
